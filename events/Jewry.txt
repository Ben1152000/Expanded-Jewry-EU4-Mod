country_event = { # Reformation of the Sanhedrin
	id = expanded_jewry.1
	title = expanded_jewry.1.t
	desc = expanded_jewry.1.d
	picture = REFORM_eventPicture

	trigger = { 
		is_year = 1530
		NOT = { is_year = 1550 }
		NOT = { religion = jewish }
		NOT = { tag = KOJ }
		NOT = { culture = al_suryah_arabic }
		owns = 379
		OR = {
			owns = 364 
			owns = 1854
		}
	}

	mean_time_to_happen = {
		months = 60
		
	}

	option = {
		name = expanded_jewry.1.a
		ai_chance = { factor = 100 }
		add_prestige = 5
		379 = {
			add_base_tax = -1
			add_base_production = -1
			spawn_rebels = {
				type = religious_rebels
				religion = jewish
				size = 2
			}
		}	
	}
	option = {
		name = expanded_jewry.1.b
		379 = {
			add_local_autonomy = 50
		}
	}
	option = {
		name = expanded_jewry.1.c
		379 = {
			add_base_tax = 1
			add_base_production = 1
			change_culture = expanded_jewry_sephardic
			change_religion = jewish
		}
		
	}

}
